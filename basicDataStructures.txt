// array methods
array.length;
array.push()   array.pop()       array.unshift()   array.shift()
arr.splice(1, 2, 'apple', 'banana'); // start at index 1, remove 2 elements, replace it with 'apple', 'banana'
arr.slice(1, 3); // copies elements from index [1, 3) into a new array
newArr.push([...arr]); // spread operator makes a copy of the entire array
arr.indexOf(elem); // returns index of element in array or -1 if it does not exist

// objects
// see basicDataStructures.txt
// similar to arrays but use properties to access them instead
var cat = {
  "name": "Whiskers",
  "legs": 4,
  "enemies": ["Water", "Dogs"]
};

// dot notation vs bracket notation
var x = cat.name; // Whiskers
var y = cat["legs"]; // 4

// methods
cat.name = "felix"; // updating object property
cat["sound"] = "meow"; // add new property
delete cat.sound; // delete property
cat.hasOwnProperty("weight"); // check if property exists
"legs" in cat; // check if property exists
Object.keys(cat); // returns array of keys
